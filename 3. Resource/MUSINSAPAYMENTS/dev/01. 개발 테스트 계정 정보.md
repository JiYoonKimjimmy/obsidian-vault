```sql
-- Sandbox
select  
    mc.hash_ci  
    , mc.customer_uid  
    , mmc.member_uid  
from member.customer mc  
inner join money.money_customer mmc on mc.hash_ci = mmc.ci  
where mc.hash_ci = '590fc6148e51d4e0846459c8ed2332c31d7d607306e5bf570c01b741904ae644'  
order by mc.rt desc  
;
```
