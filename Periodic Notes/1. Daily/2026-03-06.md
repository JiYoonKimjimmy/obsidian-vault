

|            | AFFILIATED_PROCESSING | AFFILIATED_COMPLETED | OPEN_BANK_AGREE_PROCESSING | COMPLETED        |
| ---------- | --------------------- | -------------------- | -------------------------- | ---------------- |
| PROCESSING | 케이뱅크 계좌 개설 필요         | 오픈뱅킹 등록 진행 필요        | 오픈뱅킹 등록 진행 필요              | -                |
| LINKED     | -                     | -                    | -                          | 케이뱅크 계좌 연동 완료    |
| UNLINKED   | -                     | -                    | -                          | 기존 케이뱅크 계좌 연동 필요 |

- 케이뱅크 계좌 자체 해지되는 경우는?
	- 기존 케이뱅크 계좌 재연결 시, 케이뱅크 계좌 정보 조회하여 계좌 상태 `TERMINATED` 인 경우, 에러 응답 처리

---

```sql
create table kbank_customer_master
(
    hash_ci     varchar(500)                       not null comment '고객 hash ci' primary key,
    customer_di varchar(600)                       not null comment '고객 di (hash_ci + salt)',
    rt          datetime default CURRENT_TIMESTAMP not null comment '등록일시'
) comment 'K뱅크 고객 마스터 테이블';

create table money_kbank.kbank_customer
(
    hash_ci           varchar(500) not null comment '고객 hash ci',
    account_encrypted varchar(100) null comment '케이뱅크 계좌번호(암호화)',
    main_pay_no       varchar(30)  null comment '대표 계좌 정보',
    account_id        varchar(40)  null comment '케이뱅크에서 생성한 계좌 식별자',
    process_status    varchar(50)  null comment '진행 상태 (AFFILIATED_PROCESSING, AFFILIATED_COMPLETED, OPEN_BANK_AGREE_PROCESSING, COMPLETED)',
    status            varchar(20)  null comment '연동 상태 (PROCESSING, LINKED, UNLINKED)',
    ut                datetime     null comment '수정 일시',
    rt                datetime     null comment '등록일시',
    constraint uk_kc_hash_ci_account_encrypted unique (hash_ci, account_encrypted)
) comment 'kbank 연동 정보';

create table money_kbank.kbank_customer_history
(
    seq               bigint auto_increment comment '시퀀스' primary key,
    hash_ci           varchar(500)                       not null comment '고객 hash ci',
    account_encrypted varchar(100)                       null comment '케이뱅크 계좌번호(암호화)',
    main_pay_no       varchar(30)                        null comment '대표 계좌 정보',
    account_id        varchar(40)                        null comment '케이뱅크에서 생성한 계좌 식별자',
    process_status    varchar(50)                        null comment '진행 상태 (AFFILIATED_PROCESSING, AFFILIATED_COMPLETED, OPEN_BANK_AGREE_PROCESSING, COMPLETED)',
    status            varchar(20)                        null comment '연동 상태 (PROCESSING, LINKED, UNLINKED)',
    rt                datetime default CURRENT_TIMESTAMP not null comment '등록일시'
) comment 'kbank 연동 히스토리';

create index money_kbank.idx_kch_hash_ci on money_kbank.kbank_customer_history (hash_ci);
```
